// Orientierungs-Fix für AR.js let currentOrientation = window.orientation || 0; function handleOrientationChange() { const newOrientation = window.orientation || 0; if (currentOrientation !== newOrientation) { currentOrientation = newOrientation; // Kurze Verzögerung für Layout-Neuberechnung setTimeout(() => { const scene = document.querySelector('a-scene'); if (scene && scene.renderer) { scene.renderer.setSize(window.innerWidth, window.innerHeight); } // Kamera neu ausrichten const arSystem = scene.systems['arjs']; if (arSystem && arSystem.videoElement) { const video = arSystem.videoElement; video.style.transform = getVideoTransform(currentOrientation); } }, 300); } } function getVideoTransform(orientation) { const isIOS = /iPad|iPhone|iPod/.test(navigator.userAgent); if (isIOS) { // iOS braucht meist keine Rotation return 'translate(-50%, -50%)'; } // Android Orientierungs-Anpassungen switch(orientation) { case 0: // Portrait return 'translate(-50%, -50%) rotate(0deg)'; case 90: // Landscape left return 'translate(-50%, -50%) rotate(-90deg)'; case -90: // Landscape right return 'translate(-50%, -50%) rotate(90deg)'; case 180: // Portrait upside down return 'translate(-50%, -50%) rotate(180deg)'; default: return 'translate(-50%, -50%)'; } } // Event Listener window.addEventListener('orientationchange', handleOrientationChange); window.addEventListener('resize', handleOrientationChange); // Initial Setup nach Scene-Load document.querySelector('a-scene').addEventListener('loaded', function() { handleOrientationChange(); });
